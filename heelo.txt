    <script>
        var flipCount = 0;
        var texts = ["","","One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten"];
        var flipInterval = setInterval(function() {
            var card = document.querySelector('.card');
            card.classList.toggle('flipped');
            flipCount++;

            setTimeout(function() {
                var frontFace = document.querySelector('.front');
                var backFace = document.querySelector('.back');
                frontFace.textContent = texts[flipCount +1];
                backFace.textContent = texts[flipCount + 1];
            },500); // Half of the transition duration

            if (flipCount >= Infinity) {
                clearInterval(flipInterval);
            }
        }, 1000);
    </script>
